blackAI = function(data){
    for(var i=0; i<19; i++){
        for(var j=0; j<19; j++){
            if(data.map[i][j].flag == 0){
                return {i:i, j:j};
            }
        }
    }
}

whiteAI = function(data){
    for(var i=0; i<19; i++){
        for(var j=0; j<19; j++){
            if(data.map[i][j].flag == 0){
                return {i:i, j:j};
            }
        }
    }
}
